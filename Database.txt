-- 1. Create the Database
CREATE DATABASE IF NOT EXISTS script_database;
USE script_database;

-- 2. Create Users Table
CREATE TABLE IF NOT EXISTS users (
    id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(50) NOT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 3. Create Main SF1 Records Table
-- This table is created before SF4 because SF4 relies on the LRN from this table.
CREATE TABLE IF NOT EXISTS sf1_records (
    id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    lrn VARCHAR(20) NOT NULL UNIQUE,
    name VARCHAR(100) NOT NULL,
    gender VARCHAR(10) DEFAULT NULL,
    birthdate VARCHAR(50) DEFAULT NULL,
    address VARCHAR(255) DEFAULT NULL,
    section VARCHAR(50) DEFAULT 'Earth',
    grade_level VARCHAR(20) DEFAULT 'Grade 7',
    admission_date VARCHAR(50) DEFAULT NULL,
    strand VARCHAR(50) DEFAULT '',
    profile_picture LONGBLOB DEFAULT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 4. Create SF1 Archives Table
-- This mirrors the main SF1 table perfectly to prevent column count errors during archiving.
CREATE TABLE IF NOT EXISTS sf1_records_archive (
    archive_id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    lrn VARCHAR(20) DEFAULT NULL,
    name VARCHAR(100) DEFAULT NULL,
    gender VARCHAR(10) DEFAULT NULL,
    birthdate VARCHAR(50) DEFAULT NULL,
    address VARCHAR(255) DEFAULT NULL,
    grade_level VARCHAR(20) DEFAULT NULL,
    section VARCHAR(50) DEFAULT NULL,
    admission_date VARCHAR(50) DEFAULT NULL,
    strand VARCHAR(50) DEFAULT '',
    profile_picture LONGBLOB DEFAULT NULL,
    date_archived TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 5. Create SF4 Records Table
-- Includes the Foreign Key linking to sf1_records and the Unique Key to prevent duplicate months.
CREATE TABLE IF NOT EXISTS sf4_records (
    id INT(11) NOT NULL AUTO_INCREMENT PRIMARY KEY,
    lrn VARCHAR(20) NOT NULL,
    attendance_month VARCHAR(20) NOT NULL,
    attendance_year INT(11) NOT NULL,
    requirements VARCHAR(255) DEFAULT '',
    status VARCHAR(50) DEFAULT 'Enrolled',
    UNIQUE KEY unique_monthly (lrn, attendance_month, attendance_year),
    CONSTRAINT fk_sf4_lrn FOREIGN KEY (lrn) REFERENCES sf1_records(lrn) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
