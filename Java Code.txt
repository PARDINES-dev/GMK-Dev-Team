class DatabaseHelper {
    static final String DB_URL = "jdbc:mysql://localhost:3306/script_database";
    static final String DB_USER = "root";
    static final String DB_PASS = "";

    static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(DB_URL, DB_USER, DB_PASS);
    }

    static void setupDatabase() {
        try (Connection conn = getConnection(); Statement stmt = conn.createStatement()) {
            stmt.execute("CREATE TABLE IF NOT EXISTS users (" +
                    "id INT AUTO_INCREMENT PRIMARY KEY, username VARCHAR(50) UNIQUE NOT NULL, password VARCHAR(50) NOT NULL)");
            
            stmt.execute("CREATE TABLE IF NOT EXISTS sf1_records (" +
                    "id INT AUTO_INCREMENT PRIMARY KEY, lrn VARCHAR(20) UNIQUE NOT NULL, name VARCHAR(100) NOT NULL, " +
                    "gender VARCHAR(10), birthdate VARCHAR(50), address VARCHAR(255))");

            try { stmt.execute("ALTER TABLE sf1_records ADD COLUMN section VARCHAR(50) DEFAULT 'Earth'"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf1_records ADD COLUMN grade_level VARCHAR(20) DEFAULT 'Grade 7'"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf1_records ADD COLUMN admission_date VARCHAR(50)"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf1_records ADD COLUMN strand VARCHAR(50) DEFAULT ''"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf1_records ADD COLUMN profile_picture LONGBLOB"); } catch (SQLException ignored) { }
            
            // Create Archive Table to match the main table
            stmt.execute("CREATE TABLE IF NOT EXISTS sf1_records_archive (" +
                    "archive_id INT AUTO_INCREMENT PRIMARY KEY, " +
                    "lrn VARCHAR(20), name VARCHAR(100), gender VARCHAR(10), birthdate VARCHAR(50), address VARCHAR(255), " +
                    "grade_level VARCHAR(20), section VARCHAR(50), admission_date VARCHAR(50), strand VARCHAR(50), " +
                    "profile_picture LONGBLOB, archived_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP)");

            // --- AUTOMATIC DATABASE PATCHES ---
            // These lines force MySQL to safely add missing columns to your archive table if they don't exist yet, 
            // completely fixing the "Unknown column 'strand'" crash automatically.
            try { stmt.execute("ALTER TABLE sf1_records_archive ADD COLUMN strand VARCHAR(50) DEFAULT ''"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf1_records_archive ADD COLUMN profile_picture LONGBLOB"); } catch (SQLException ignored) { }

            stmt.execute("CREATE TABLE IF NOT EXISTS sf4_records (" +
                    "id INT AUTO_INCREMENT PRIMARY KEY, " +
                    "lrn VARCHAR(20) NOT NULL, " +
                    "attendance_month VARCHAR(20) NOT NULL, " +
                    "attendance_year INT NOT NULL, " +
                    "requirements VARCHAR(255) DEFAULT '', " +
                    "status VARCHAR(50) DEFAULT 'Enrolled', " +
                    "UNIQUE KEY unique_monthly (lrn, attendance_month, attendance_year), " +
                    "FOREIGN KEY (lrn) REFERENCES sf1_records(lrn) ON DELETE CASCADE ON UPDATE CASCADE)");

            try { stmt.execute("ALTER TABLE sf4_records ADD COLUMN requirements VARCHAR(255) DEFAULT ''"); } catch (SQLException ignored) { }
            try { stmt.execute("ALTER TABLE sf4_records ADD COLUMN status VARCHAR(50) DEFAULT 'Enrolled'"); } catch (SQLException ignored) { }
                    
        } catch (SQLException e) {
            System.out.println("Database Error: " + e.getMessage());
        }
    }
}

class AuthPanel extends JPanel {
    private CardLayout authLayout = new CardLayout();

    public AuthPanel(SCRIPT_PROJECT mainFrame) {
        this.mainFrame = mainFrame;

    private JPanel createLoginCard() {
        JPanel p = new JPanel(new GridBagLayout());
        p.setBackground(Color.WHITE);
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(10, 10, 10, 10);
        gbc.fill = GridBagConstraints.HORIZONTAL;

        JLabel lblTitle = new JLabel("Login", SwingConstants.CENTER);
        lblTitle.setFont(new Font("Arial", Font.BOLD, 26));
        JTextField txtUser = new JTextField(20);
        txtUser.setFont(new Font("Arial", Font.PLAIN, 16));
        JPasswordField txtPass = new JPasswordField(20);
        txtPass.setFont(new Font("Arial", Font.PLAIN, 16));
        JButton btnLogin = new JButton("Login");
        JButton btnToReg = new JButton("Create New Account");

        btnLogin.setBackground(SCRIPT_PROJECT.HEADER_COLOR);
        btnLogin.setForeground(Color.WHITE);
        btnLogin.setFont(new Font("Arial", Font.BOLD, 14));

        btnToReg.addActionListener(e -> authLayout.show(authContainer, "REGISTER"));
        btnLogin.addActionListener(e -> {
            String u = txtUser.getText();
            String p_word = new String(txtPass.getPassword());
            try (Connection conn = DatabaseHelper.getConnection()) {
                PreparedStatement pst = conn.prepareStatement("SELECT * FROM users WHERE username=? AND password=?");
                pst.setString(1, u);
                pst.setString(2, p_word);
                if (pst.executeQuery().next()) {
                    txtUser.setText("");
                    txtPass.setText("");
                    mainFrame.showDashboard();
                } else {
                    JOptionPane.showMessageDialog(this, "Invalid credentials.");
                }
            } catch (SQLException ex) {
                JOptionPane.showMessageDialog(this, "Database Connection Failed.");
            }
        });

        gbc.gridy=0; p.add(lblTitle, gbc);
        gbc.gridy=1; p.add(new JLabel("Username:"), gbc);
        gbc.gridy=2; p.add(txtUser, gbc);
        gbc.gridy=3; p.add(new JLabel("Password:"), gbc);
        gbc.gridy=4; p.add(txtPass, gbc);
        gbc.gridy=5; p.add(btnLogin, gbc);
        gbc.gridy=6; p.add(btnToReg, gbc);
        return p;
    }

        gbc.gridy=5; p.add(btnReg, gbc);

                "Are you sure you want to archive this record? It will be moved to the archive database.", 
                "Confirm Archive", 
                JOptionPane.YES_NO_OPTION);

            if (confirm == JOptionPane.YES_OPTION) {
                String lrnToArchive = sf1Table.getValueAt(row, 1).toString(); // Index 1 is LRN

                try (Connection conn = DatabaseHelper.getConnection()) {
                    // Step 1: Copy to Archive
                    String archiveSql = "INSERT INTO sf1_records_archive (lrn, name, gender, birthdate, address, grade_level, section, admission_date, strand, profile_picture) " +
                                        "SELECT lrn, name, gender, birthdate, address, grade_level, section, admission_date, strand, profile_picture FROM sf1_records WHERE lrn = ?";
                    try (PreparedStatement insertPst = conn.prepareStatement(archiveSql)) {
                        insertPst.setString(1, lrnToArchive);
                        insertPst.executeUpdate();
                    }

                    // Step 2: Delete from main table
                    String deleteSql = "DELETE FROM sf1_records WHERE lrn = ?";
                    try (PreparedStatement deletePst = conn.prepareStatement(deleteSql)) {
                        deletePst.setString(1, lrnToArchive);
                        deletePst.executeUpdate();
                    }

                    JOptionPane.showMessageDialog(this, "Record successfully archived.", "Success", JOptionPane.INFORMATION_MESSAGE);
                    loadSF1Data(); 
                } catch (SQLException ex) {
                    ex.printStackTrace();
                    JOptionPane.showMessageDialog(this, "Database Error: " + ex.getMessage(), "Error", JOptionPane.ERROR_MESSAGE);
                }
            }
        });

    private void loadSF4Data(String gradeFilter, String sectionFilter, String monthFilter, String yearFilter, String lrnFilter, String admissionDateFilter) {
        
        // AUTOMATIC DATE FALLBACK FOR SQL QUERY
        if (monthFilter == null || monthFilter.trim().isEmpty() || yearFilter == null || yearFilter.trim().isEmpty()) {
            Calendar cal = Calendar.getInstance();
            monthFilter = new SimpleDateFormat("MMMM").format(cal.getTime());
            yearFilter = String.valueOf(cal.get(Calendar.YEAR));
        }

        sf4Model.setRowCount(0);
        String query = "SELECT s.profile_picture, s.lrn, s.name, s.grade_level, s.section, s.admission_date, " +
                       "IFNULL(a.requirements, '') AS requirements, IFNULL(a.status, 'Enrolled') AS status " +
                       "FROM sf1_records s " +
                       "LEFT JOIN sf4_records a ON s.lrn = a.lrn AND a.attendance_month = ? AND a.attendance_year = ? " +
                       "WHERE 1=1";
                       
        if (!gradeFilter.equals("All Grades")) query += " AND s.grade_level = ?";
        if (!sectionFilter.equals("All Sections")) query += " AND s.section = ?";
        if (!lrnFilter.isEmpty()) query += " AND (s.lrn LIKE ? OR s.name LIKE ?)";
        if (!admissionDateFilter.isEmpty()) query += " AND s.admission_date = ?"; 
        
        try (Connection conn = DatabaseHelper.getConnection();
             PreparedStatement pst = conn.prepareStatement(query)) {
             
            int pIndex = 1;
            pst.setString(pIndex++, monthFilter);
            pst.setInt(pIndex++, yearFilter.isEmpty() ? 2026 : Integer.parseInt(yearFilter));
            
            if (!gradeFilter.equals("All Grades")) pst.setString(pIndex++, gradeFilter);
            if (!sectionFilter.equals("All Sections")) pst.setString(pIndex++, sectionFilter);
            if (!lrnFilter.isEmpty()) {
                pst.setString(pIndex++, "%" + lrnFilter + "%");
                pst.setString(pIndex++, "%" + lrnFilter + "%");
            }
            if (!admissionDateFilter.isEmpty()) pst.setString(pIndex++, admissionDateFilter);

            ResultSet rs = pst.executeQuery();

            while(rs.next()) {
                byte[] photo = rs.getBytes("profile_picture");
                String rawReqs = rs.getString("requirements");
                String grade = rs.getString("grade_level");
                
                boolean isSHS = grade.equals("Grade 11") || grade.equals("Grade 12");
                int maxReqs = isSHS ? 4 : 3;
                
                String reqDisplay = "Pending";
                if(rawReqs != null && !rawReqs.isEmpty()) {
                    int completedCount = rawReqs.split(",").length;
                    if(completedCount >= maxReqs) reqDisplay = "Completed";
                    else reqDisplay = completedCount + "/" + maxReqs + " Submitted";
                }

                sf4Model.addRow(new Object[]{
                    photo, rs.getString("lrn"), rs.getString("name"), rs.getString("grade_level"), 
                    rs.getString("section"), rs.getString("admission_date"), 
                    reqDisplay, rs.getString("status") 
                });
            }
        } catch (SQLException ex) {
            System.out.println("Error Loading SF4: " + ex.getMessage());
        }
    }

    private void showRequirementsUI(String lrn, String name, String grade, String month, String year) {
        boolean isSHS = grade.equals("Grade 11") || grade.equals("Grade 12");
        
        String[] reqList = isSHS ?
            new String[]{"Certificate of junior High School Completion or Diploma", "Certificate of Good Moral Character", "PSA Birth Certificate", "2x2photo"} :
            new String[]{"PSA birth Certificate", "Form 138", "2x2 picture"};
            
        String savedReqs = "";
        try (Connection conn = DatabaseHelper.getConnection()) {
            PreparedStatement pst = conn.prepareStatement("SELECT requirements FROM sf4_records WHERE lrn=? AND attendance_month=? AND attendance_year=?");
            pst.setString(1, lrn); 
            pst.setString(2, month); 
            pst.setInt(3, Integer.parseInt(year));
            ResultSet rs = pst.executeQuery();
            if (rs.next()) savedReqs = rs.getString("requirements");
        } catch (Exception ex) { 
            ex.printStackTrace();
        }

        JPanel panel = new JPanel(new GridLayout(reqList.length + 1, 1, 5, 5));
        panel.add(new JLabel("Check submitted documents for " + name + " (" + grade + "):"));
        
        JCheckBox[] boxes = new JCheckBox[reqList.length];
        for (int i = 0; i < reqList.length; i++) {
            boxes[i] = new JCheckBox(reqList[i]);
            if (savedReqs != null && savedReqs.contains(reqList[i])) {
                boxes[i].setSelected(true);
            }
            panel.add(boxes[i]);
        }

        int option = JOptionPane.showConfirmDialog(null, panel, "Requirements Checklist", JOptionPane.OK_CANCEL_OPTION);
        if (option == JOptionPane.OK_OPTION) {
            StringBuilder newReqs = new StringBuilder();
            for (JCheckBox box : boxes) {
                if (box.isSelected()) {
                    newReqs.append(box.getText()).append(",");
                }
            }
            
            try (Connection conn = DatabaseHelper.getConnection()) {
                String sql = "INSERT INTO sf4_records (lrn, attendance_month, attendance_year, requirements, status) " +
                             "VALUES (?, ?, ?, ?, 'Enrolled') ON DUPLICATE KEY UPDATE requirements = VALUES(requirements)";
                PreparedStatement pst = conn.prepareStatement(sql);
                pst.setString(1, lrn);
                pst.setString(2, month);
                pst.setInt(3, Integer.parseInt(year));
                pst.setString(4, newReqs.toString());
                pst.executeUpdate();
                
                String formattedDate = "";
                if(dcSF4DateFilter.getDate() != null) {
                    formattedDate = new SimpleDateFormat("MMMM dd, yyyy").format(dcSF4DateFilter.getDate());
                }
                loadSF4Data(cbGrade.getSelectedItem().toString(), cbSection.getSelectedItem().toString(), month, year, txtLRNFilter.getText().trim(), formattedDate);
            } catch (Exception ex) {
                JOptionPane.showMessageDialog(this, "Error saving requirements: " + ex.getMessage());
            }
        }
    }

    private void updateStudentStatus(String lrn, String month, String year, String status) {
        try (Connection conn = DatabaseHelper.getConnection()) {
            String sql = "INSERT INTO sf4_records (lrn, attendance_month, attendance_year, requirements, status) " +
                         "VALUES (?, ?, ?, '', ?) ON DUPLICATE KEY UPDATE status = VALUES(status)";
            PreparedStatement pst = conn.prepareStatement(sql);
            pst.setString(1, lrn);
            pst.setString(2, month);
            pst.setInt(3, Integer.parseInt(year));
            pst.setString(4, status);
            pst.executeUpdate();
        } catch (Exception ex) {
            System.out.println("Error saving status: " + ex.getMessage());
        }
    }

    private void saveRecord() {
        if (txtLRN.getText().trim().isEmpty() || 
            txtFirstName.getText().trim().isEmpty() || 
            txtLastName.getText().trim().isEmpty() || 
            txtMiddleName.getText().trim().isEmpty() || 
            txtAddress.getText().trim().isEmpty() || 
            dcBirthdate.getDate() == null || 
            dcAdmission.getDate() == null || 
            selectedImageFile == null) { 
            
            JOptionPane.showMessageDialog(this, 
                "Please fill all required fields marked with (*), including the 2x2 photo attachment.", 
                "Validation Error", 
                JOptionPane.ERROR_MESSAGE);
            return;
        }

        String lrn = txtLRN.getText().trim();
        String mName = txtMiddleName.getText().trim();
        String fullName = txtFirstName.getText().trim() + (mName.isEmpty() ? "" : " " + mName) + " " + txtLastName.getText().trim();
        
        SimpleDateFormat sdf = new SimpleDateFormat("MMMM dd, yyyy");
        String formattedBirthdate = sdf.format(dcBirthdate.getDate());
        String formattedAdmission = sdf.format(dcAdmission.getDate());
        
        String strandVal = "";
        String selectedGrade = cbGrade.getSelectedItem().toString();
        if (selectedGrade.equals("Grade 11") || selectedGrade.equals("Grade 12")) {
            strandVal = cbStrand.getSelectedItem().toString();
        }

        if(lrn.length() != 12) {
            JOptionPane.showMessageDialog(this, "LRN must be exactly 12 digits.");
            return;
        }

        try (Connection conn = DatabaseHelper.getConnection()) {
            PreparedStatement pst = conn.prepareStatement(
                "INSERT INTO sf1_records (profile_picture, lrn, name, gender, birthdate, address, grade_level, section, admission_date, strand) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)"
            );
            
            FileInputStream fis = new FileInputStream(selectedImageFile);
            pst.setBinaryStream(1, fis, (int) selectedImageFile.length());
            
            pst.setString(2, lrn);
            pst.setString(3, fullName);
            pst.setString(4, cbGender.getSelectedItem().toString());
            pst.setString(5, formattedBirthdate); 
            pst.setString(6, txtAddress.getText());
            pst.setString(7, selectedGrade);
            pst.setString(8, cbSection.getSelectedItem().toString());
            pst.setString(9, formattedAdmission); 
            pst.setString(10, strandVal); 
            
            pst.executeUpdate();
            parentPanel.loadSF1Data();
            dispose();
        } catch (Exception ex) {
            JOptionPane.showMessageDialog(this, "Error saving record: Ensure LRN is unique.");
        }
    }

            ResultSet rs = pst.executeQuery();
            if (rs.next()) {
                txtFullName.setText(rs.getString("name"));
                cbGender.setSelectedItem(rs.getString("gender"));
                txtAddress.setText(rs.getString("address"));

                String grade = rs.getString("grade_level");
                cbGrade.setSelectedItem(grade);
                SectionData.updateDropdown(grade, cbSection, false);
                cbSection.setSelectedItem(rs.getString("section"));

                if (grade.equals("Grade 11") || grade.equals("Grade 12")) {
                    cbStrand.setEnabled(true);
                    cbStrand.setSelectedItem(rs.getString("strand"));
                } else {
                    cbStrand.setEnabled(false);
                }

                SimpleDateFormat sdf = new SimpleDateFormat("MMMM dd, yyyy");
                try {
                    String bday = rs.getString("birthdate");
                    if (bday != null && !bday.isEmpty()) dcBirthdate.setDate(sdf.parse(bday));

                    String adm = rs.getString("admission_date");
                    if (adm != null && !adm.isEmpty()) dcAdmission.setDate(sdf.parse(adm));
                } catch (Exception parseEx) {
                    System.out.println("Date parsing format logic error passed safely.");
                }
            }
        } catch (Exception ex) {
            ex.printStackTrace();
        }
    }

    private void updateRecord() {
        String fullName = txtFullName.getText().trim();
        SimpleDateFormat sdf = new SimpleDateFormat("MMMM dd, yyyy");

        String formattedBirthdate = dcBirthdate.getDate() != null ? sdf.format(dcBirthdate.getDate()) : "";
        String formattedAdmission = dcAdmission.getDate() != null ? sdf.format(dcAdmission.getDate()) : "";

        String selectedGrade = cbGrade.getSelectedItem().toString();
        String strandVal = (selectedGrade.equals("Grade 11") || selectedGrade.equals("Grade 12")) ? cbStrand.getSelectedItem().toString() : "";

        try (Connection conn = DatabaseHelper.getConnection()) {
            PreparedStatement pst = conn.prepareStatement(
                "UPDATE sf1_records SET name=?, gender=?, birthdate=?, address=?, grade_level=?, section=?, admission_date=?, strand=? WHERE lrn=?"
            );
            pst.setString(1, fullName);
            pst.setString(2, cbGender.getSelectedItem().toString());
            pst.setString(3, formattedBirthdate);
            pst.setString(4, txtAddress.getText());
            pst.setString(5, selectedGrade);
            pst.setString(6, cbSection.getSelectedItem().toString());
            pst.setString(7, formattedAdmission);
            pst.setString(8, strandVal);
            pst.setString(9, originalLrn);

            pst.executeUpdate();
            parentPanel.loadSF1Data();
            dispose();
        } catch (Exception ex) {
            JOptionPane.showMessageDialog(this, "Error updating record.");
        }
    }
}

// 7. CUSTOM IMAGE RENDERER FOR JTABLE
// ==========================================
class ImageRenderer extends DefaultTableCellRenderer {
    private int width, height;

    public ImageRenderer(int width, int height) {
        this.width = width;
        this.height = height;
    }

    @Override
    public Component getTableCellRendererComponent(JTable table, Object value, boolean isSelected, boolean hasFocus, int row, int column) {
        JLabel label = new JLabel();
        label.setHorizontalAlignment(JLabel.CENTER);

        if (value != null && value instanceof byte[]) {
            byte[] bytes = (byte[]) value;
            if (bytes.length > 0) {
                ImageIcon icon = new ImageIcon(bytes);
                Image img = icon.getImage().getScaledInstance(width, height, Image.SCALE_SMOOTH);
                label.setIcon(new ImageIcon(img));
            } else {
                label.setText("No Photo");
            }
        } else {
            label.setText("No Photo");
        }

        if (isSelected) {
            label.setBackground(table.getSelectionBackground());
            label.setForeground(table.getSelectionForeground());
            label.setOpaque(true);
        } else {
            label.setBackground(table.getBackground());
            label.setForeground(table.getForeground());
            label.setOpaque(true);
        }
        return label;
    }
}

// 8. HELPER CLASS FOR DYNAMIC SECTIONS

class SectionData {
    public static void updateDropdown(String grade, JComboBox<String> cbSection, boolean includeAll) {
        cbSection.removeAllItems();
        if (includeAll) {
            cbSection.addItem("All Sections");
        }
        
        String[] sections;
        switch (grade) {
            case "Grade 7":
                sections = new String[]{"Mercury", "Venus", "Earth", "Mars", "Jupiter", "Saturn", "Uranus", "Neptune", "Pluto"};
                break;
            case "Grade 8":
                sections = new String[]{"Rose", "Sampaguita", "Jasmine", "Orchid", "Sunflower", "Daisy", "Tulip", "Lily", "Carnation"};
                break;
            case "Grade 9":
                sections = new String[]{"Rizal", "Bonifacio", "Mabini", "Luna", "Del Pilar", "Aguinaldo", "Jacinto", "Silang", "Malvar"};
                break;
            case "Grade 10":
                sections = new String[]{"Diamond", "Emerald", "Ruby", "Sapphire", "Amethyst", "Garnet", "Topaz", "Opal", "Pearl"};
                break;
            case "Grade 11":
            case "Grade 12":
                sections = new String[]{"Block 1", "Block 2", "Block 3", "Block 4", "Block 5", "Block 6", "Block 7", "Block 8", "Block 9"};
                break;
            default:
                sections = new String[]{};
                break;
        }
        
        for (String s : sections) {
            cbSection.addItem(s);
        }
    }
}
